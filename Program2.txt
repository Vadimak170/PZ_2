int             GT(int & a, int & b)
{
    return a > b;
}
void LINEARINSERTION(int a[], unsigned long n)
{
    unsigned long   i,
                    j;
    int           tmp;

    for (i = 1; i < n; i++)
        for (j = i; j > 0 && GT(a[j - 1], a[j]); j--) {

            tmp = a[j];
            a[j] = a[j - 1];
            a[j - 1] = tmp;
        }
}

void selection(int *ar, int size)
 {
     int i, j, mx, nmx;

     for(i = 0; i < size-1; i++)
       {
         mx=ar[i];
         nmx=i;
         for(j = i+1; j < size; j++)
           {
             if (ar[j]<mx)
                {
                 mx=ar[j];
                 nmx=j;
                 }
           }
          ar[nmx]=ar[i];
          ar[i]=mx;
           }
        cout<<"Массив отсортирован методом выбора.\n";
     }
     void GnomeSort(int *A, int N) {
    int i = 0;
    while(i < N) {
        if(i == 0 || A[i - 1] <= A[i]) ++i;
        else {
            int Temp = A[i];
            A[i] = A[i - 1];
            A[i - 1] = Temp;
            --i;
        }
    }
}
